debugfs=off                       # Disables the debug filesystem.

l1tf=full,force                   # Enables full mitigation of the L1 Terminal Fault (L1TF) vulnerability.
lockdown=confidentiality          # Enables the Linux kernel lockdown mode in confidentiality mode.
loglevel=0                        # Sets the kernel log level to 0 (only emergency messages).

nosmt=force                       # Forces the disabling of simultaneous multithreading (SMT).
oops=panic                        # Configures the system to panic on kernel oops.

page_poison=on                    # Enables poisoning of freed pages to catch use-after-free bugs.
pti=on                            # Enables Page Table Isolation (PTI) to mitigate the Meltdown vulnerability.
quiet                             # Reduces the amount of kernel log messages output during boot.
randomize_kstack_offset=on        # Randomizes the kernel stack offset at each system call.
rng_core.default_quality=500      # Sets the default quality of the random number generator to 500, affecting the entropy estimation.
slab_nomerge=yes                  # Prevents the merging of slab caches.
slub_debug=FZP                    # Enables debugging options for the SLUB allocator: F (free), Z (zeroing of objects), and P (poisoning of objects).
spec_store_bypass_disable=on      # Disables speculative store bypass to mitigate certain side-channel attacks.
spec_store_bypass_disable=seccomp # Disables speculative store bypass but allows seccomp to selectively enable it for certain processes.
